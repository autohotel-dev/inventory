create view public.products_view as
select
  p.id,
  p.name,
  p.sku,
  p.description,
  p.price,
  p.cost,
  p.min_stock,
  p.unit,
  p.barcode,
  p.category_id,
  c.name as category_name,
  p.subcategory_id,
  s.name as subcategory_name,
  p.supplier_id,
  sup.name as supplier_name,
  p.is_active,
  p.created_at,
  p.updated_at,
  COALESCE(inv.total_stock, 0::numeric) as total_stock,
  COALESCE(inv.total_stock, 0::numeric) * p.price as inventory_value,
  case
    when COALESCE(inv.total_stock, 0::numeric) <= 0::numeric then 'critical'::text
    when COALESCE(inv.total_stock, 0::numeric) < p.min_stock then 'low'::text
    when COALESCE(inv.total_stock, 0::numeric) >= (p.min_stock * 3::numeric) then 'high'::text
    else 'normal'::text
  end as stock_status
from
  products p
  left join categories c on p.category_id = c.id
  left join subcategories s on p.subcategory_id = s.id
  left join suppliers sup on p.supplier_id = sup.id
  left join (
    select
      stock.product_id,
      sum(stock.qty) as total_stock
    from
      stock
    group by
      stock.product_id
  ) inv on p.id = inv.product_id;
